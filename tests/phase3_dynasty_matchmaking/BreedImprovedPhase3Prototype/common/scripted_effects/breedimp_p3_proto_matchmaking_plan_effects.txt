breedimp_p3_proto_preflight_fail_effect = {
	set_variable = {
		name = breedimp_p3_proto_failure_slot
		value = $SLOT$
	}
	set_variable = {
		name = breedimp_p3_proto_failure_reason
		value = flag:$REASON$
	}
}

breedimp_p3_proto_preflight_validate_slot_effect = {
	if = {
		limit = {
			NOT = { exists = var:breedimp_p3_proto_failure_reason }
			NOT = {
				breedimp_p3_proto_slot_is_valid_or_empty_trigger = { SLOT = $SLOT$ }
			}
		}
		breedimp_p3_proto_preflight_fail_effect = {
			SLOT = $SLOT$
			REASON = breedimp_p3_proto_failure_invalid_slot
		}
	}
}

breedimp_p3_proto_preflight_compare_pair_effect = {
	if = {
		limit = {
			NOT = { exists = var:breedimp_p3_proto_failure_reason }
			NOT = {
				breedimp_p3_proto_slot_pair_has_no_collision_trigger = {
					LEFT = $LEFT$
					RIGHT = $RIGHT$
				}
			}
		}
		breedimp_p3_proto_preflight_fail_effect = {
			SLOT = $RIGHT$
			REASON = breedimp_p3_proto_failure_duplicate_character
		}
	}
}

breedimp_p3_proto_execute_slot_effect = {
	# Dispatch uses only the already preflighted stored enums.
	if = {
		limit = {
			breedimp_p3_proto_slot_is_committed_trigger = { SLOT = $SLOT$ }
		}
		if = {
			limit = {
				var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_marriage
				var:breedimp_p3_proto_slot_$SLOT$_direction = flag:breedimp_p3_proto_direction_ordinary
			}
			breedimp_p3_proto_create_ordinary_marriage_effect = {
				SUBJECT = root.var:breedimp_p3_proto_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_p3_proto_slot_$SLOT$_partner
			}
		}
		else_if = {
			limit = {
				var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_marriage
				var:breedimp_p3_proto_slot_$SLOT$_direction = flag:breedimp_p3_proto_direction_matrilineal
			}
			breedimp_p3_proto_create_matrilineal_marriage_effect = {
				SUBJECT = root.var:breedimp_p3_proto_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_p3_proto_slot_$SLOT$_partner
			}
		}
		else_if = {
			limit = {
				var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_betrothal
				var:breedimp_p3_proto_slot_$SLOT$_direction = flag:breedimp_p3_proto_direction_ordinary
			}
			breedimp_p3_proto_create_ordinary_betrothal_effect = {
				SUBJECT = root.var:breedimp_p3_proto_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_p3_proto_slot_$SLOT$_partner
			}
		}
		else = {
			breedimp_p3_proto_create_matrilineal_betrothal_effect = {
				SUBJECT = root.var:breedimp_p3_proto_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_p3_proto_slot_$SLOT$_partner
			}
		}
	}
}
