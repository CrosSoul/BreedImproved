# Breed Improved Phase 3 isolated prototype lifecycle and cleanup effects.
# The permanent one-workflow-per-save lock is intentionally never removed.
# Test-only, non-release, CK3 1.19.0.6. Runtime status: NOT RUN.

breedimp_p3_proto_clear_slot_effect = {
	if = {
		limit = { exists = var:breedimp_p3_proto_slot_$SLOT$_subject }
		remove_variable = breedimp_p3_proto_slot_$SLOT$_subject
	}
	if = {
		limit = { exists = var:breedimp_p3_proto_slot_$SLOT$_partner }
		remove_variable = breedimp_p3_proto_slot_$SLOT$_partner
	}
	if = {
		limit = { exists = var:breedimp_p3_proto_slot_$SLOT$_direction }
		remove_variable = breedimp_p3_proto_slot_$SLOT$_direction
	}
	if = {
		limit = { exists = var:breedimp_p3_proto_slot_$SLOT$_relationship_type }
		remove_variable = breedimp_p3_proto_slot_$SLOT$_relationship_type
	}
	if = {
		limit = { exists = var:breedimp_p3_proto_slot_$SLOT$_placeholder }
		remove_variable = breedimp_p3_proto_slot_$SLOT$_placeholder
	}
	if = {
		limit = { exists = var:breedimp_p3_proto_slot_$SLOT$_reservation_id }
		remove_variable = breedimp_p3_proto_slot_$SLOT$_reservation_id
	}
}

breedimp_p3_proto_clear_event_lists_effect = {
	every_in_list = {
		list = breedimp_p3_proto_subjects
		remove_from_list = breedimp_p3_proto_subjects
	}
	every_in_list = {
		list = breedimp_p3_proto_reviewed_subjects
		remove_from_list = breedimp_p3_proto_reviewed_subjects
	}
	every_in_list = {
		list = breedimp_p3_proto_deferred_subjects

	}
	every_in_list = {

breedimp_p3_proto_cleanup_workflow_effect = {
	breedimp_p3_proto_clear_actor_state_effect = yes

	if = {
		limit = { exists = global_var:breedimp_p3_proto_phase }
		remove_global_variable = breedimp_p3_proto_phase
	}
	if = {
		limit = { exists = global_var:breedimp_p3_proto_active_actor }
		remove_global_variable = breedimp_p3_proto_active_actor
	}
}

breedimp_p3_proto_cleanup_active_workflow_effect = {
	if = {
		limit = { exists = global_var:breedimp_p3_proto_active_actor }
		global_var:breedimp_p3_proto_active_actor = {
			breedimp_p3_proto_cleanup_workflow_effect = yes
		}
	}
	else = {
		if = {
			limit = { exists = global_var:breedimp_p3_proto_phase }
			remove_global_variable = breedimp_p3_proto_phase
		}
	}
}
