breedimp_p3_proto_slot_is_empty_trigger = {
	NOT = { exists = var:breedimp_p3_proto_slot_$SLOT$_subject }
	NOT = { exists = var:breedimp_p3_proto_slot_$SLOT$_partner }
	NOT = { exists = var:breedimp_p3_proto_slot_$SLOT$_direction }
	NOT = { exists = var:breedimp_p3_proto_slot_$SLOT$_relationship_type }
	NOT = { exists = var:breedimp_p3_proto_slot_$SLOT$_placeholder }

breedimp_p3_proto_slot_is_committed_trigger = {
	exists = var:breedimp_p3_proto_slot_$SLOT$_subject
	exists = var:breedimp_p3_proto_slot_$SLOT$_partner

	exists = var:breedimp_p3_proto_slot_$SLOT$_relationship_type
	exists = var:breedimp_p3_proto_slot_$SLOT$_placeholder
	exists = var:breedimp_p3_proto_slot_$SLOT$_reservation_id
	var:breedimp_p3_proto_slot_$SLOT$_reservation_id = var:breedimp_p3_proto_slot_$SLOT$_subject
	OR = {
		var:breedimp_p3_proto_slot_$SLOT$_direction = flag:breedimp_p3_proto_direction_ordinary
		var:breedimp_p3_proto_slot_$SLOT$_direction = flag:breedimp_p3_proto_direction_matrilineal
	}
	OR = {
		var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_marriage
		var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_betrothal
	}
	var:breedimp_p3_proto_slot_$SLOT$_placeholder = flag:breedimp_p3_proto_placeholder_none
	NOT = {
		var:breedimp_p3_proto_slot_$SLOT$_subject = var:breedimp_p3_proto_slot_$SLOT$_partner
	}
}

breedimp_p3_proto_slot_relationship_type_matches_ages_trigger = {
	OR = {
		AND = {
			var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_marriage
			var:breedimp_p3_proto_slot_$SLOT$_subject = { is_adult = yes }
			var:breedimp_p3_proto_slot_$SLOT$_partner = { is_adult = yes }
		}
		AND = {
			var:breedimp_p3_proto_slot_$SLOT$_relationship_type = flag:breedimp_p3_proto_relationship_betrothal
			OR = {
				var:breedimp_p3_proto_slot_$SLOT$_subject = { is_adult = no }
				var:breedimp_p3_proto_slot_$SLOT$_partner = { is_adult = no }
			}
		}
	}
}
