breedimp_dynasty_matchmaking_rejection_is_committed_trigger = {
	exists = var:breedimp_dynasty_matchmaking_rejection_$SLOT$_subject
	exists = var:breedimp_dynasty_matchmaking_rejection_$SLOT$_partner
	exists = var:breedimp_dynasty_matchmaking_rejection_$SLOT$_marker
	var:breedimp_dynasty_matchmaking_rejection_$SLOT$_marker = var:breedimp_dynasty_matchmaking_rejection_$SLOT$_subject
}

breedimp_dynasty_matchmaking_rejection_matches_pair_trigger = {
	breedimp_dynasty_matchmaking_rejection_is_committed_trigger = { SLOT = $SLOT$ }
	OR = {
		AND = {
			var:breedimp_dynasty_matchmaking_rejection_$SLOT$_subject = $SUBJECT$
			var:breedimp_dynasty_matchmaking_rejection_$SLOT$_partner = $PARTNER$
		}
		AND = {
			var:breedimp_dynasty_matchmaking_rejection_$SLOT$_subject = $PARTNER$
			var:breedimp_dynasty_matchmaking_rejection_$SLOT$_partner = $SUBJECT$
		}
	}
}

breedimp_dynasty_matchmaking_character_base_eligible_trigger = {
	is_alive = yes
	is_ai = yes
	is_available = yes
	is_travelling = no
	is_incapable = no
	is_married = no
	is_betrothed = no
	is_imprisoned = no
	is_hostage = no
	is_concubine = no
	is_pregnant = no
	is_pool_guest = no
	NOT = { has_trait = celibate }
	exists = dynasty
	dynasty = $ACTOR$.var:breedimp_dynasty_matchmaking_managed_dynasty
	NOT = { this = $ACTOR$ }
}

breedimp_dynasty_matchmaking_subject_candidate_trigger = {
	breedimp_dynasty_matchmaking_character_base_eligible_trigger = { ACTOR = $ACTOR$ }
	OR = {
		is_adult = no
		fertility > 0
	}
	breedimp_dynasty_matchmaking_character_is_unreserved_trigger = { ACTOR = $ACTOR$ }
}

breedimp_dynasty_matchmaking_normal_partner_candidate_trigger = {
	breedimp_dynasty_matchmaking_character_base_eligible_trigger = { ACTOR = $ACTOR$ }
	OR = {
		is_adult = no
		fertility > 0
	}
	NOT = { is_in_list = breedimp_dynasty_matchmaking_skipped_subjects }
	NOT = { is_in_list = breedimp_dynasty_matchmaking_deferred_subjects }
	breedimp_dynasty_matchmaking_character_is_unreserved_trigger = { ACTOR = $ACTOR$ }
}

breedimp_dynasty_matchmaking_placeholder_partner_candidate_trigger = {
	breedimp_dynasty_matchmaking_character_base_eligible_trigger = { ACTOR = $ACTOR$ }
	is_adult = yes
	fertility <= 0
	NOT = { is_in_list = breedimp_dynasty_matchmaking_skipped_subjects }
	NOT = { is_in_list = breedimp_dynasty_matchmaking_deferred_subjects }
	breedimp_dynasty_matchmaking_character_is_unreserved_trigger = { ACTOR = $ACTOR$ }
}

breedimp_dynasty_matchmaking_subject_allows_placeholder_trigger = {
	is_adult = yes

		always = yes
	}
}

breedimp_dynasty_matchmaking_pair_age_policy_trigger = {
	NOT = {
		AND = {
			is_adult = no
			$PARTNER$ = {
				OR = {
					AND = { is_female = yes age >= 30 }
					AND = { is_male = yes age >= 40 }
				}
			}
		}
	}
	NOT = {
		AND = {
			$PARTNER$ = { is_adult = no }
			OR = {
				AND = { is_female = yes age >= 30 }
				AND = { is_male = yes age >= 40 }
			}
		}
	}
}
