breedimp_dynasty_matchmaking_kinship_score_value = {
	value = 4
	if = {
		limit = { is_close_family_of = scope:comparator }
		subtract = 4
	}
	else_if = {
		limit = {
			OR = {
				is_uncle_or_aunt_of = scope:comparator
				is_nibling_of = scope:comparator
			}
		}
		subtract = 3
	}
	else_if = {
		limit = { is_cousin_of = scope:comparator }
		subtract = 2
	}
	else_if = {
		limit = { is_extended_family_of = scope:comparator }
		subtract = 1
	}
}

# Lexicographic order: age gap, congenital score, kinship category, then a
# deterministic script-visible character fingerprint. Engine list order remains
# the final fallback only when every exposed component is exactly tied.

	value = age_difference

	add = {
		value = breedimp_dynasty_matchmaking_kinship_score_value
		multiply = 1000
	}
	add = breedimp_dynasty_matchmaking_stable_tiebreak_value
}
