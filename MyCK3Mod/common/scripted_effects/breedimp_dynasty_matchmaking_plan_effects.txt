breedimp_dynasty_matchmaking_save_plan_anchor_effect = {
	if = {
		limit = {
			breedimp_dynasty_matchmaking_slot_is_committed_trigger = { SLOT = 1 }
		}
		var:breedimp_dynasty_matchmaking_slot_1_subject = {
			save_scope_as = breedimp_dynasty_matchmaking_plan_anchor_subject
		}
		var:breedimp_dynasty_matchmaking_slot_1_partner = {
			save_scope_as = breedimp_dynasty_matchmaking_plan_anchor_partner
		}
	}
}

breedimp_dynasty_matchmaking_fail_effect = {
	set_variable = {
		name = breedimp_dynasty_matchmaking_failure_slot
		value = $SLOT$
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_failure_reason
		value = flag:$REASON$
	}
}

breedimp_dynasty_matchmaking_preflight_validate_slot_effect = {
	if = {
		limit = {
			NOT = { exists = var:breedimp_dynasty_matchmaking_failure_reason }
			NOT = {
				breedimp_dynasty_matchmaking_slot_is_valid_or_empty_trigger = { SLOT = $SLOT$ }
			}
		}
		breedimp_dynasty_matchmaking_fail_effect = {
			SLOT = $SLOT$
			REASON = breedimp_dynasty_matchmaking_failure_invalid_slot
		}
	}
	else_if = {
		limit = {
			NOT = { exists = var:breedimp_dynasty_matchmaking_failure_reason }
			breedimp_dynasty_matchmaking_slot_is_committed_trigger = { SLOT = $SLOT$ }
			OR = {
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_subject = {
					is_in_list = breedimp_dynasty_matchmaking_preflight_participants
				}
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_partner = {
					is_in_list = breedimp_dynasty_matchmaking_preflight_participants
				}
			}
		}
		breedimp_dynasty_matchmaking_fail_effect = {
			SLOT = $SLOT$
			REASON = breedimp_dynasty_matchmaking_failure_duplicate_character
		}
	}
	else_if = {
		limit = {
			NOT = { exists = var:breedimp_dynasty_matchmaking_failure_reason }
			breedimp_dynasty_matchmaking_slot_is_committed_trigger = { SLOT = $SLOT$ }
		}
		var:breedimp_dynasty_matchmaking_slot_$SLOT$_subject = {
			add_to_list = breedimp_dynasty_matchmaking_preflight_participants
		}
		var:breedimp_dynasty_matchmaking_slot_$SLOT$_partner = {
			add_to_list = breedimp_dynasty_matchmaking_preflight_participants
		}
		change_variable = {
			name = breedimp_dynasty_matchmaking_preflight_pair_count
			add = 1
		}
	}
}

breedimp_dynasty_matchmaking_execute_slot_effect = {
	if = {
		limit = {
			breedimp_dynasty_matchmaking_workflow_is_executing_trigger = yes
			NOT = { exists = var:breedimp_dynasty_matchmaking_execution_failed }
			breedimp_dynasty_matchmaking_slot_is_committed_trigger = { SLOT = $SLOT$ }
		}

		if = {
			limit = {
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type = flag:breedimp_dynasty_matchmaking_relationship_marriage
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_direction = flag:breedimp_dynasty_matchmaking_direction_ordinary
			}
			breedimp_dynasty_matchmaking_create_ordinary_marriage_effect = {
				SUBJECT = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_partner
			}
		}
		else_if = {
			limit = {
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type = flag:breedimp_dynasty_matchmaking_relationship_marriage
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_direction = flag:breedimp_dynasty_matchmaking_direction_matrilineal
			}
			breedimp_dynasty_matchmaking_create_matrilineal_marriage_effect = {
				SUBJECT = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_partner
			}
		}
		else_if = {
			limit = {
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type = flag:breedimp_dynasty_matchmaking_relationship_betrothal
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_direction = flag:breedimp_dynasty_matchmaking_direction_ordinary
			}
			breedimp_dynasty_matchmaking_create_ordinary_betrothal_effect = {
				SUBJECT = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_partner
			}
		}
		else_if = {
			limit = {
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type = flag:breedimp_dynasty_matchmaking_relationship_betrothal
				var:breedimp_dynasty_matchmaking_slot_$SLOT$_direction = flag:breedimp_dynasty_matchmaking_direction_matrilineal
			}
			breedimp_dynasty_matchmaking_create_matrilineal_betrothal_effect = {
				SUBJECT = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_subject
				PARTNER = root.var:breedimp_dynasty_matchmaking_slot_$SLOT$_partner
			}
		}
		else = {
			set_variable = {
				name = breedimp_dynasty_matchmaking_execution_failed
				value = flag:breedimp_dynasty_matchmaking_execution_failed
			}
			breedimp_dynasty_matchmaking_fail_effect = {
				SLOT = $SLOT$
				REASON = breedimp_dynasty_matchmaking_failure_invalid_slot
			}
		}

		if = {
			limit = {
				NOT = { exists = var:breedimp_dynasty_matchmaking_execution_failed }
				breedimp_dynasty_matchmaking_slot_relationship_exists_trigger = { SLOT = $SLOT$ }
			}
			change_variable = {
				name = breedimp_dynasty_matchmaking_executed_pair_count
				add = 1
			}
			if = {
				limit = {
					var:breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type = flag:breedimp_dynasty_matchmaking_relationship_marriage
				}
				change_variable = {
					name = breedimp_dynasty_matchmaking_executed_marriage_count
					add = 1
				}
			}
			else = {
				change_variable = {
					name = breedimp_dynasty_matchmaking_executed_betrothal_count
					add = 1
				}
			}
		}
		else_if = {
			limit = {
				NOT = { exists = var:breedimp_dynasty_matchmaking_execution_failed }
			}
			set_variable = {
				name = breedimp_dynasty_matchmaking_execution_failed
				value = flag:breedimp_dynasty_matchmaking_execution_failed
			}
			breedimp_dynasty_matchmaking_fail_effect = {
				SLOT = $SLOT$
				REASON = breedimp_dynasty_matchmaking_failure_relationship_not_created
			}
		}
	}
}

breedimp_dynasty_matchmaking_confirm_plan_unchecked_effect = {
	breedimp_dynasty_matchmaking_perform_full_preflight_effect = yes

	if = {
		limit = {
			exists = var:breedimp_dynasty_matchmaking_preflight_passed
			var:breedimp_dynasty_matchmaking_preflight_passed = flag:breedimp_dynasty_matchmaking_preflight_passed
		}
		set_global_variable = {
			name = breedimp_dynasty_matchmaking_phase
			value = flag:breedimp_dynasty_matchmaking_phase_executing
		}
		breedimp_dynasty_matchmaking_execute_validated_plan_effect = yes

		if = {
			limit = { exists = var:breedimp_dynasty_matchmaking_execution_failed }
			set_global_variable = {
				name = breedimp_dynasty_matchmaking_phase
				value = flag:breedimp_dynasty_matchmaking_phase_failed
			}
			breedimp_dynasty_matchmaking_trigger_partial_failure_event_effect = yes
		}
		else = {
			set_global_variable = {
				name = breedimp_dynasty_matchmaking_phase
				value = flag:breedimp_dynasty_matchmaking_phase_completed
			}
			breedimp_dynasty_matchmaking_trigger_success_event_effect = yes
		}
	}
	else = {
		set_global_variable = {
			name = breedimp_dynasty_matchmaking_phase
			value = flag:breedimp_dynasty_matchmaking_phase_failed
		}
		breedimp_dynasty_matchmaking_trigger_preflight_failure_event_effect = yes
	}
}
