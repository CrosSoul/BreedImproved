breedimp_dynasty_matchmaking_clear_partner_list_effect = {
	every_in_list = {
		list = breedimp_dynasty_matchmaking_partner_candidates
		remove_from_list = breedimp_dynasty_matchmaking_partner_candidates
	}
	if = {
		limit = { exists = var:breedimp_dynasty_matchmaking_current_placeholder }
		remove_variable = breedimp_dynasty_matchmaking_current_placeholder
	}
	if = {
		limit = { exists = var:breedimp_dynasty_matchmaking_displayed_subject }
		remove_variable = breedimp_dynasty_matchmaking_displayed_subject
	}
	if = {
		limit = { exists = var:breedimp_dynasty_matchmaking_displayed_partner }
		remove_variable = breedimp_dynasty_matchmaking_displayed_partner
	}
	if = {
		limit = { exists = var:breedimp_dynasty_matchmaking_displayed_subject_fertility_percent }
		remove_variable = breedimp_dynasty_matchmaking_displayed_subject_fertility_percent
	}
	if = {
		limit = { exists = var:breedimp_dynasty_matchmaking_displayed_partner_fertility_percent }
		remove_variable = breedimp_dynasty_matchmaking_displayed_partner_fertility_percent
	}
}

breedimp_dynasty_matchmaking_trigger_review_event_effect = {
	set_variable = {
		name = breedimp_dynasty_matchmaking_displayed_subject
		value = scope:breedimp_dynasty_matchmaking_current_subject
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_displayed_partner
		value = scope:breedimp_dynasty_matchmaking_current_partner
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_displayed_subject_fertility_percent
		value = {
			value = scope:breedimp_dynasty_matchmaking_current_subject.fertility
			multiply = 100
		}
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_displayed_partner_fertility_percent
		value = {
			value = scope:breedimp_dynasty_matchmaking_current_partner.fertility
			multiply = 100
		}
	}

	if = {
		limit = {
			var:breedimp_dynasty_matchmaking_current_token = flag:breedimp_dynasty_matchmaking_token_a
		}
		trigger_event = breedimp_dynasty_matchmaking.2020
	}
	else_if = {
		limit = {
			var:breedimp_dynasty_matchmaking_current_token = flag:breedimp_dynasty_matchmaking_token_b
		}
		trigger_event = breedimp_dynasty_matchmaking.2021
	}
}

breedimp_dynasty_matchmaking_build_subject_list_effect = {
	var:breedimp_dynasty_matchmaking_managed_dynasty = {
		every_dynasty_member = {
			limit = {
				breedimp_dynasty_matchmaking_subject_candidate_trigger = { ACTOR = root }
			}
			add_to_list = breedimp_dynasty_matchmaking_subjects
			root = {
				change_variable = {
					name = breedimp_dynasty_matchmaking_subject_count
					add = 1
				}
			}
		}
	}

	breedimp_dynasty_matchmaking_select_next_subject_effect = yes
}

breedimp_dynasty_matchmaking_build_partner_list_effect = {
	breedimp_dynasty_matchmaking_clear_partner_list_effect = yes

	var:breedimp_dynasty_matchmaking_managed_dynasty = {
		every_dynasty_member = {
			limit = {
				breedimp_dynasty_matchmaking_character_base_eligible_trigger = { ACTOR = root }
				breedimp_dynasty_matchmaking_character_is_unreserved_trigger = { ACTOR = root }
			}
			save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
			if = {
				limit = {
					scope:breedimp_dynasty_matchmaking_current_subject = {
						breedimp_dynasty_matchmaking_pair_core_is_eligible_trigger = {
							ACTOR = root
							PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
						}
					}
			}
			add_to_list = breedimp_dynasty_matchmaking_partner_candidates
		}
	}
	}

	breedimp_dynasty_matchmaking_select_partner_for_current_subject_effect = yes
}

breedimp_dynasty_matchmaking_select_next_subject_effect = {
	if = {
		limit = {
			any_in_list = {
				list = breedimp_dynasty_matchmaking_subjects
				NOT = { is_in_list = breedimp_dynasty_matchmaking_reviewed_subjects }
				breedimp_dynasty_matchmaking_subject_candidate_trigger = { ACTOR = root }
			}
		}
		ordered_in_list = {
			list = breedimp_dynasty_matchmaking_subjects
			limit = {
				NOT = { is_in_list = breedimp_dynasty_matchmaking_reviewed_subjects }
				breedimp_dynasty_matchmaking_subject_candidate_trigger = { ACTOR = root }
			}
			order_by = breedimp_dynasty_matchmaking_subject_order_value
			max = 1
			check_range_bounds = no
			save_scope_as = breedimp_dynasty_matchmaking_current_subject
			add_to_list = breedimp_dynasty_matchmaking_reviewed_subjects
		}
		breedimp_dynasty_matchmaking_build_partner_list_effect = yes
	}
	else_if = {
		limit = {
			any_in_list = {
				list = breedimp_dynasty_matchmaking_deferred_subjects
				breedimp_dynasty_matchmaking_subject_candidate_trigger = { ACTOR = root }
			}
		}
		ordered_in_list = {
			list = breedimp_dynasty_matchmaking_deferred_subjects
			limit = {
				breedimp_dynasty_matchmaking_subject_candidate_trigger = { ACTOR = root }
			}
			order_by = breedimp_dynasty_matchmaking_subject_order_value
			max = 1
			check_range_bounds = no
			save_scope_as = breedimp_dynasty_matchmaking_current_subject
			remove_from_list = breedimp_dynasty_matchmaking_deferred_subjects
		}
		breedimp_dynasty_matchmaking_build_partner_list_effect = yes
	}
	else = {
		if = {
			limit = { breedimp_dynasty_matchmaking_plan_has_committed_pair_trigger = yes }
			breedimp_dynasty_matchmaking_prepare_final_confirmation_effect = yes
		}
		else_if = {
			limit = { var:breedimp_dynasty_matchmaking_subject_count > 0 }
			breedimp_dynasty_matchmaking_cleanup_workflow_effect = yes
			trigger_event = breedimp_dynasty_matchmaking.2181
		}
		else = {
			breedimp_dynasty_matchmaking_cleanup_workflow_effect = yes
			trigger_event = breedimp_dynasty_matchmaking.2180
		}
	}
}

breedimp_dynasty_matchmaking_select_partner_for_current_subject_effect = {
	scope:breedimp_dynasty_matchmaking_current_subject = {
		save_scope_as = comparator
	}

	if = {
		limit = {
			any_in_list = {
				list = breedimp_dynasty_matchmaking_partner_candidates
				save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
				scope:breedimp_dynasty_matchmaking_current_subject = {
					breedimp_dynasty_matchmaking_normal_pair_is_eligible_trigger = {
						ACTOR = root
						PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
					}
				}
			}
		}
		if = {
			limit = {
				scope:breedimp_dynasty_matchmaking_current_subject = { is_adult = yes }
			}
			ordered_in_list = {
				list = breedimp_dynasty_matchmaking_partner_candidates
				limit = {
					save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
					scope:breedimp_dynasty_matchmaking_current_subject = {
						breedimp_dynasty_matchmaking_normal_pair_is_eligible_trigger = {
							ACTOR = root
							PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
						}
					}
				}
				order_by = breedimp_dynasty_matchmaking_fertility_order_value
				max = 1
				check_range_bounds = no
				save_scope_as = breedimp_dynasty_matchmaking_best_fertility_partner
			}
			ordered_in_list = {
				list = breedimp_dynasty_matchmaking_partner_candidates
				limit = {
					save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
					scope:breedimp_dynasty_matchmaking_current_subject = {
						breedimp_dynasty_matchmaking_normal_pair_is_eligible_trigger = {
							ACTOR = root
							PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
						}
					}
					fertility >= {
						value = scope:breedimp_dynasty_matchmaking_best_fertility_partner.fertility
						subtract = 0.05
					}
					fertility <= scope:breedimp_dynasty_matchmaking_best_fertility_partner.fertility
				}
				order_by = breedimp_dynasty_matchmaking_partner_rank_value
				max = 1
				check_range_bounds = no
				save_scope_as = breedimp_dynasty_matchmaking_current_partner
			}
		}
		else = {
			ordered_in_list = {
				list = breedimp_dynasty_matchmaking_partner_candidates
				limit = {
					save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
					scope:breedimp_dynasty_matchmaking_current_subject = {
						breedimp_dynasty_matchmaking_normal_pair_is_eligible_trigger = {
							ACTOR = root
							PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
						}
					}
				}
				order_by = breedimp_dynasty_matchmaking_partner_rank_value
				max = 1
				check_range_bounds = no
				save_scope_as = breedimp_dynasty_matchmaking_current_partner
			}
		}
		set_variable = {
			name = breedimp_dynasty_matchmaking_current_placeholder
			value = flag:breedimp_dynasty_matchmaking_placeholder_none
		}
		breedimp_dynasty_matchmaking_trigger_review_event_effect = yes
	}
	else_if = {
		limit = {
			scope:breedimp_dynasty_matchmaking_current_subject = {
				breedimp_dynasty_matchmaking_subject_allows_placeholder_trigger = yes
			}
			any_in_list = {
				list = breedimp_dynasty_matchmaking_partner_candidates
				save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
				scope:breedimp_dynasty_matchmaking_current_subject = {
					breedimp_dynasty_matchmaking_placeholder_pair_is_eligible_trigger = {
						ACTOR = root
						PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
					}
				}
			}
		}
		ordered_in_list = {
			list = breedimp_dynasty_matchmaking_partner_candidates
			limit = {
				save_temporary_scope_as = breedimp_dynasty_matchmaking_partner_candidate_temp
				scope:breedimp_dynasty_matchmaking_current_subject = {
					breedimp_dynasty_matchmaking_placeholder_pair_is_eligible_trigger = {
						ACTOR = root
						PARTNER = scope:breedimp_dynasty_matchmaking_partner_candidate_temp
					}
				}
			}
			order_by = breedimp_dynasty_matchmaking_partner_rank_value
			max = 1
			check_range_bounds = no
			save_scope_as = breedimp_dynasty_matchmaking_current_partner
		}
		set_variable = {
			name = breedimp_dynasty_matchmaking_current_placeholder
			value = flag:breedimp_dynasty_matchmaking_placeholder_zero_fertility
		}
		breedimp_dynasty_matchmaking_trigger_review_event_effect = yes
	}
	else = {
		breedimp_dynasty_matchmaking_select_next_subject_effect = yes
	}
}

breedimp_dynasty_matchmaking_commit_slot_effect = {
	set_variable = {
		name = breedimp_dynasty_matchmaking_slot_$SLOT$_subject
		value = scope:breedimp_dynasty_matchmaking_current_subject
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_slot_$SLOT$_partner
		value = scope:breedimp_dynasty_matchmaking_current_partner
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_slot_$SLOT$_direction
		value = flag:$DIRECTION$
	}
	if = {
		limit = {
			scope:breedimp_dynasty_matchmaking_current_subject = { is_adult = yes }
			scope:breedimp_dynasty_matchmaking_current_partner = { is_adult = yes }
		}
		set_variable = {
			name = breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type
			value = flag:breedimp_dynasty_matchmaking_relationship_marriage
		}
	}
	else = {
		set_variable = {
			name = breedimp_dynasty_matchmaking_slot_$SLOT$_relationship_type
			value = flag:breedimp_dynasty_matchmaking_relationship_betrothal
		}
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_slot_$SLOT$_placeholder
		value = var:breedimp_dynasty_matchmaking_current_placeholder
	}

	# Commit marker is deliberately written last.
	set_variable = {
		name = breedimp_dynasty_matchmaking_slot_$SLOT$_reservation_id
		value = scope:breedimp_dynasty_matchmaking_current_subject
	}
	change_variable = {
		name = breedimp_dynasty_matchmaking_accepted_pair_count
		add = 1
	}
}
