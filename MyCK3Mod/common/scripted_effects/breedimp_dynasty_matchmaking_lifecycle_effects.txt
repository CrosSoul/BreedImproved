breedimp_dynasty_matchmaking_cleanup_workflow_effect = {
	breedimp_dynasty_matchmaking_clear_actor_state_effect = yes
	if = {
		limit = { exists = global_var:breedimp_dynasty_matchmaking_phase }
		remove_global_variable = breedimp_dynasty_matchmaking_phase
	}
	if = {
		limit = {
			exists = global_var:breedimp_dynasty_matchmaking_active_actor
			global_var:breedimp_dynasty_matchmaking_active_actor = this
		}
		remove_global_variable = breedimp_dynasty_matchmaking_active_actor
	}
}

breedimp_dynasty_matchmaking_activate_workflow_unchecked_effect = {
	breedimp_dynasty_matchmaking_clear_actor_state_effect = yes

	if = {
		limit = { has_character_flag = breedimp_dynasty_matchmaking_next_token_b }
		remove_character_flag = breedimp_dynasty_matchmaking_next_token_b
		set_variable = {
			name = breedimp_dynasty_matchmaking_current_token
			value = flag:breedimp_dynasty_matchmaking_token_b
		}
	}
	else = {
		add_character_flag = breedimp_dynasty_matchmaking_next_token_b
		set_variable = {
			name = breedimp_dynasty_matchmaking_current_token
			value = flag:breedimp_dynasty_matchmaking_token_a
		}
	}

	set_global_variable = {
		name = breedimp_dynasty_matchmaking_active_actor
		value = root
	}
	set_global_variable = {
		name = breedimp_dynasty_matchmaking_phase
		value = flag:breedimp_dynasty_matchmaking_phase_review
	}
	set_variable = {
		name = breedimp_dynasty_matchmaking_managed_dynasty
		value = root.dynasty
	}
	set_variable = { name = breedimp_dynasty_matchmaking_accepted_pair_count value = 0 }
	set_variable = { name = breedimp_dynasty_matchmaking_executed_pair_count value = 0 }
	set_variable = { name = breedimp_dynasty_matchmaking_executed_marriage_count value = 0 }
	set_variable = { name = breedimp_dynasty_matchmaking_executed_betrothal_count value = 0 }
	set_variable = { name = breedimp_dynasty_matchmaking_subject_count value = 0 }

	breedimp_dynasty_matchmaking_build_subject_list_effect = yes
}

breedimp_dynasty_matchmaking_activate_workflow_effect = {
	if = {
		limit = {
			breedimp_dynasty_matchmaking_actor_can_activate_trigger = yes
		}
		breedimp_dynasty_matchmaking_activate_workflow_unchecked_effect = yes
	}
}

	if = {
		limit = {
			breedimp_dynasty_matchmaking_actor_owns_active_workflow_trigger = yes
		}
		breedimp_dynasty_matchmaking_cleanup_workflow_effect = yes
		trigger_event = breedimp_dynasty_matchmaking.2190
	}
